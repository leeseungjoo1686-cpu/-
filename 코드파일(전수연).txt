import streamlit as st
import google.generativeai as genai
import requests
from bs4 import BeautifulSoup 

# 1. API 키 설정 (본인의 Gemini API 키를 입력하세요)
# 실제 서비스 시에는 st.secrets 등을 활용해 안전하게 관리하는 것이 좋습니다.
GOOGLE_API_KEY = "YOUR_GEMINI_API_KEY_HERE" 
genai.configure(api_key=GOOGLE_API_KEY) 

# 2. 웹사이트에서 텍스트 추출하는 함수 (함수실행 시 약간의 오류 발생으로 수정 필요) 
def extract_text_from_url(url):
    try:
        headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'}
        response = requests.get(url, headers=headers, timeout=10)
        
        if response.status_code == 200:
            soup = BeautifulSoup(response.text, 'html.parser')
            # 불필요한 스크립트나 스타일 제거
            for script in soup(["script", "style", "nav", "footer", "header"]):
                script.decompose()
            # 본문 텍스트 추출 및 정제
            text = soup.get_text()
            lines = (line.strip() for line in text.splitlines())
            chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
            cleaned_text = '\n'.join(chunk for chunk in chunks if chunk)
            return cleaned_text[:4000] # LLM 입력 제한을 고려해 앞부분 4000자만 추출
        else:
            return f"Error: 웹사이트에 접근할 수 없습니다. (상태 코드: {response.status_code})"
    except Exception as e:
        return f"Error: 자료를 가져오는 중 오류가 발생했습니다. ({str(e)})" 

# 3. Streamlit 웹 UI 구성
st.set_page_config(page_title="AI 자료 사이트 분석기", layout="wide") 

st.title("🌐 AI 웹 자료 분석기")
st.subtitle("분석하고 싶은 사이트의 URL을 입력하면 AI가 핵심을 요약하고 분석해 드립니다.") 

# 입력창 및 조건 설정
url_input = st.text_input("분석할 사이트 URL 주소를 입력하세요:", placeholder="https://example.com") 

# 분석 프롬프트 (핵심 대화 틀)
analysis_option = st.selectbox(
    "분석 스타일을 선택하세요:",
    ["핵심 요약 및 3줄 요약", "비판적 분석 및 한계점 도출", "트렌드 및 인사이트 추출", "쉬운 설명 (초보자용)"]
) 

if st.button("분석 시작하기 ✨"):
    if not url_input:
        st.warning("URL을 입력해 주세요!")
    elif GOOGLE_API_KEY == "YOUR_GEMINI_API_KEY_HERE":
        st.error("API 키가 올바르지 않습니다. 코드에서 생성한 API 키를 넣어주세요.")
    else:
        with st.spinner("사이트에서 자료를 읽어와 분석하는 중입니다..."):
            # 1단계: 웹 스크래핑
            site_content = extract_text_from_url(url_input)
            
            if site_content.startswith("Error"):
                st.error(site_content)
            else:
                # 2단계: 프롬프트 구성 (바이브코딩 핵심)
                prompt = f"""
                너는 전문 데이터 분석가이자 리서처야. 
                아래 제공된 웹사이트 본문 내용을 기반으로 요구사항에 맞춰 분석해줘.
                
                [분석 요구사항]
                {analysis_option} 스타일로 작성해줘. 
                답변은 가독성이 좋게 마크다운(Markdown) 형태로 제목과 불릿포인트(•)를 적절히 사용해줘.
                
                [웹사이트 본문 내용]
                {site_content}
                """
                
                try:
                    # 3단계: Gemini 모델로 요청 발송
                    model = genai.GenerativeModel('gemini-1.5-flash') # 빠르고 효율적인 모델
                    response = model.generate_content(prompt)
                    
                    # 4단계: 결과 화면 출력
                    st.success("✅ 분석 완료!")
                    
                    col1, col2 = st.columns(2)
                    with col1:
                        st.subheader("📋 수집된 원본 데이터 (일부)")
                        st.text_area("텍스트 데이터", site_content, height=400)
                    with col2:
                        st.subheader("💡 AI 프롬프트 분석 결과")
                        st.markdown(response.text)
                        
                except Exception as e:
                    st.error(f"AI 분석 중 오류가 발생했습니다: {e}")
#실행시  로딩이 오래걸림, 안티그래비티와코덱스로 수정중
